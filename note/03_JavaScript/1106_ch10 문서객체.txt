1. 문서 객체 생성
	방법1) 1. h1태그 생성
			변수명1 = document.creatElement('h1');
	         2. textNode생성
			변수명2 = document.creatTextNode('h1태그 안에 추가할 문구');
	         3. textNode를 h1에 연결
			변수명1.appendChild(변수명2);
	         4. h1을 body에 연결
			document.body.appendChild(변수명1);
	
	방법2) 1. img태그 생성
	         2. src 속성 변경
			img.setAttribute('변경할 속성', '변경할 내용);
	         3. img를 body에 연결

	방법3) innerHTML 이용
		document.body.innerHTML += '추가할 내용';
