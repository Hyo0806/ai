01. 배열과 객체의 출력(속성 추가/삭제)
	*for-in문으로 출력
	for(let key in obj){
		document.writeln(`${key} : ${obj[key]}); }
	*객체 속성 추가
		obj.key = 'value';
	*객체 속성 삭제
		delete(obj.key);

02. 객체의 속성과 메소드
	this : 내 객체의 ( 파이썬의 self)
	this.key 이런식으로 사용

03. toString
	'toString' : function(){
		let output = '';
		for(let key in this){
			if (key != '메소드' && key != 'toString'){
				output += key + ':' + this[key]; }}
		return output; }
