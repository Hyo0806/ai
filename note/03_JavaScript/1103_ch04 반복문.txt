arr.push() : 배열의 끝방에 값을 추가 == 파이썬 append
arr.pop() : 제일 끝방의 값을 return하고 없앰

*반복문
	for(idx ; 반복할 조건 ; 증감식){}
	for(var idx in arr){} : object나 array에서 사용
	for(var data of arr){} : object나 array에서 사용
	arr.forEach(function(data,idx){}) : array만 사용
	arr.forEach(function(data){}) : index가 필요없을시
	
	while(조건식){} : 조건식이 참일 떄 계속 실행할 명령문들
	do {} while(조건식) : 최초 한번은 무조건실행. 두번째부터는 참일경우 계속 실행할 명령문
